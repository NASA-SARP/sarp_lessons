def convert_lon(lon_180):
    '''Convert a longitude from 0 to 360 degree range to a -180 to 180 degree range'''
    if lon_180 < 0:
        lon_360 = lon_180 + 360
    else:
        lon_360 = lon_180
    return lon_360
